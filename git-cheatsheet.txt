git status - show changed and staged files
git add FILE - stage a file for commit (replace FILE with a path)
git commit -m "MESSAGE" - save a named snapshot
git log --oneline --graph --decorate - review recent commit history

git switch -c BRANCH - create and switch to a new branch
git switch BRANCH - switch branches
git merge BRANCH - merge a branch into the current branch
git merge --abort - abort an in-progress merge (after conflicts)
git diff --name-only --diff-filter=U - list files with merge conflicts
